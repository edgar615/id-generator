DROP TABLE IF EXISTS WORKER_NODE;
CREATE TABLE WORKER_NODE (
    WORKER_NODE_ID INT NOT NULL AUTO_INCREMENT COMMENT '自增 id，不会有很多',
    NODE_HOST VARCHAR(64) NOT NULL COMMENT '主机IP，保证唯一',
    TYPE TINYINT NOT NULL COMMENT '节点类型: 1-宿主机 2-容器内',
    state TINYINT NOT NULL COMMENT '状态: 1-未用 2-占用',
    LAUNCH_TIME BIGINT NOT NULL COMMENT '启动时间',
    MODIFIED_TIME BIGINT NOT NULL COMMENT '修改时间',
    CREATED_on TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    PRIMARY KEY(ID)
) COMMENT='DB WorkerID Assigner for UID Generator',ENGINE = INNODB;
